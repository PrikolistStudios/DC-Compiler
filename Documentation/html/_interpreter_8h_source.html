335 void ProcessFunc(PolizFuncCall* func_call, int return_point) {
336 if (cur_func_.name() != "main") {
337 for (auto& val : cur_func_.local_vars()) {
338 if (!val->array_.empty()) {
339 for (auto& var : val->array_) {
340 call_stack_.top().local_vars()[var] = *var;
341 }
342 } else {
343 call_stack_.top().local_vars()[val] = *val;
344 }
345 }
346 for (auto& val : cur_func_.arguments()) {
347 if (!val->array_.empty()) {
348 for (auto& var : val->array_) {
349 call_stack_.top().local_vars()[var] = *var;
350 }
351 } else {
352 call_stack_.top().local_vars()[val] = *val;
353 }
354 }
355 }
356
357 call_stack_.emplace(return_point, func_call->func_name());
358
359 for (auto& val : funcs_) {
360 if (val.name() == func_call->func_name()) {
361 for (int i = val.arguments().size() - 1; i >= 0; --i) {
362 val.arguments()[i]->data_ = PopData();
363 }
364
365 for (auto& var : val.local_vars()) {
366 var->data_ = VarData(var->data_.type_);
367 }
368
369 cur_ind_ = val.poliz_start();
370
371 cur_func_ = val;
372
373 break;
374 }
